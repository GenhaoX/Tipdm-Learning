def incise(x,y):
    s = x.split('·')
    if(len(s)<y+1):
        return np.nan
    else:
        return s[y]

def transform(x):
    x = x.replace('人','').replace('以','').replace('上','')
    s = x.split('-')
    if(len(s)==1):
        return s[0]
    else:
        return int((int(s[0])+(int(s[1])+1))/2)

def money(x):
    if x.count('K'):
        x = x.split('K')[0]
        return (int(x.split('-')[0]) + int(x.split('-')[1]))/2*1000
    else:
        x = x.split('元')[0]
        return (int(x.split('-')[0])*30 + int(x.split('-')[1])*30)/2
