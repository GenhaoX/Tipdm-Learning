def money(x):
    if x.count('K'):
        x = x.split('K')[0]
        return (int(x.split('-')[0]) + int(x.split('-')[1]))/2*1000
    else:
        x = x.split('')[0]
        return (int(x.split('-')[0])*30 + int(x.split('-')[1])*30)/2
